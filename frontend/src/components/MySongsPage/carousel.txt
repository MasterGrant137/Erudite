import { useEffect, useState } from 'react';
import { useDispatch, useSelector } from 'react-redux';
import { mySongs } from '../../store/songs';
import './MySongsPage.css'

export const MySongsPage = () => {
    const dispatch = useDispatch();
    const iframeRegex= /(<iframe)|id="(.*?)"|src="(.*?)"|title="(.*?)"(><\/iframe>)/g

    useEffect(() => {
        dispatch(mySongs());
    }, [dispatch])

    const songs = useSelector(state => {
        return state.songs;
    })

    const [visibility, setVisibility] = useState('hidden-mySongs-lyrics');
    const [size, setSize] = useState('big-mySongs-video')


    const songAndLyricsDiv = Object.values(songs).map(song => (
        <div key={song.media.replace(iframeRegex, '$2')}>
            <iframe
                key={song.media.replace(iframeRegex, '$2')}
                id={size}
                className='mySongs-video'
                src={song.media.replace(iframeRegex,'$3')}
                title={song.title}
                allow='fullscreen'
                onMouseOver={() => {
                    setVisibility('visible-mySongs-lyrics');
                    setSize('small-mySongs-video');
                }}
                onMouseOut={() => {
                    setVisibility('hidden-mySongs-lyrics');
                    setSize('big-mySongs-video');
                }}
             />
             <textarea
                id={visibility}
                type='text'
                key={song.id}
                value={song.body}
                onMouseOver={() => {
                    setVisibility('visible-mySongs-lyrics');
                    setSize('small-mySongs-video');
                }}
                onMouseOut={() => {
                    setVisibility('hidden-mySongs-lyrics');
                    setSize('big-mySongs-video')
                }}
                disabled
                />
        </div>
    ));

    return (
        <div id='mySongs-page-container'>
            <div id='background-color-wrapper'>
                <div id='mySongs-headers-container'>
                    <div id='mySongs-page-header'>Erudite</div>
                    <div id='song-visits-header'>My Songs</div>
                </div>
                <div id='mySongs-carousel'>
                    {songAndLyricsDiv}
                </div>
            </div>
        </div>
    );
}

export default MySongsPage;
